In this program, Users enter their credit card number and the program checks if they entered a valid credit card info and if the entered info is not corect it will show an error and it wont go to the next step. 

            
import re

def main():
   user_input = inputs()
   cc_search = processing(user_input)
   outputs(user_input,cc_search)

def inputs():

    user_input = input('Please enter your credit card number with instructions on it')
    while len(user_input) < 14 :
        user_input = input('Try again, Your input should contain at least 14 charectors:')

    return user_input


def processing(user_input):



    pattern = re.compile(r'(?:\d{4}-){3}\d{4}|\d{16}|(?:\d{4}[ ]){3}\d{4}')
    cc_search = pattern.search(user_input)
    print(cc_search)
    if cc_search == None:
        print('Try again, use format xxxx-xxxx-xxxx, xxxxxxxxxx')
        main()

    else:
        cc_search = cc_search

    return user_input,cc_search


def outputs(user_input ,cc_search):
    print('Success!')
    print(f'Here\'s what you entered:  {user_input} ')
    print(f'Your credit card number is: {cc_search.group()}')

main()
